46 T random_transition(T current_state, TimeSpan dt, const std::pair<T, double> (&transitions)[NumTransitions])
47 {
48  assert(std::all_of(std::begin(transitions), std::end(transitions), [](auto& p) {
49  return p.second >= 0.0;
50  }) && "transition rates must be non-negative");
51 
52  //check if any transition happens using exponential distribution with the sum of all transition rates
53  auto sum = std::accumulate(std::begin(transitions), std::end(transitions), 0.0, [](auto&& a, auto&& t) {
54  return a + t.second;
55  });
56  if (sum <= 0) { //no transitions or all transitions have rate zero
57  return current_state;
58  }
59  auto v = ExponentialDistribution<double>::get_instance()(sum);
60  if (v < dt.days()) {

64  return t.second;
65  });

67  return transitions[random_idx].first;
