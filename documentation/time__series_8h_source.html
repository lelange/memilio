455  inline Eigen::Index next_pow2(Eigen::Index i)
456  {
457  //https://stackoverflow.com/questions/1322510/given-an-integer-how-do-i-find-the-next-largest-power-of-two-using-bit-twiddlin
458  --i;
459  i |= i >> 1;
460  i |= i >> 2;
461  i |= i >> 4;
462  i |= i >> 8;
463  i |= i >> 16;
464  IF_CONSTEXPR(sizeof(Eigen::Index) == 8) {
465  i |= i >> 32;
466  }
467  ++i;
468  return i;
469  }
