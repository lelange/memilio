37 class SecirModel : public CompartmentalModel<Populations<AgeGroup, InfectionState>, SecirParams>
38 {
39  using Base = CompartmentalModel<epi::Populations<AgeGroup, InfectionState>, SecirParams>;
40  using Pa = Base::ParameterSet;
41  using Po = Base::Populations;

43 public:
44  SecirModel(const Populations& pop, const ParameterSet& params)
45  : Base(pop, params)
46  {
47 #if !USE_DERIV_FUNC
48  size_t n_agegroups = (size_t)AgeGroup::Count;
49  for (size_t i = 0; i < n_agegroups; i++) {
50  for (size_t j = 0; j < n_agegroups; j++) {
51 
52  // Si to Ei individually for each age group j
53  this->add_flow(
54  std::make_tuple((AgeGroup)i, InfectionState::S), std::make_tuple((AgeGroup)i, InfectionState::E),
55  [i, j](Pa const& p, Eigen::Ref<const Eigen::VectorXd> pop, Eigen::Ref<const Eigen::VectorXd> y, double t) {
56 
57  //symptomatic are less well quarantined when testing and tracing is overwhelmed so they infect more people
58  auto test_and_trace_required = (1 - params.probabilities[i].get_asymp_per_infectious()) * dummy_R3 * Po::get_from(pop, (AgeGroup)i, InfectionState::C);
59  auto risk_from_symptomatic = smoother_cosine(
60  test_and_trace_required, params.get_test_and_trace_capacity(), params.get_test_and_trace_capacity() * 5,
61  params.probabilities[i].get_risk_from_symptomatic(),
62  params.probabilities[i].get_test_and_trace_max_risk_from_symptomatic());
63 
64  // effective contact rate by contact rate between groups i and j and damping j
65  ScalarType season_val =
66  (1 + p.get_seasonality() * sin(3.141592653589793 *
67  (std::fmod((p.get_start_day() + t), 365.0) / 182.5 + 0.5)));
68  ScalarType cont_freq_eff = season_val * contact_matrix.get_matrix_at(t)(static_cast<Eigen::Index>(i),
69  static_cast<Eigen::Index>(j));
70  ScalarType Nj = Po::get_from(pop, (AgeGroup)j, InfectionState::S) +
71  Po::get_from(pop, (AgeGroup)j, InfectionState::E) +
72  Po::get_from(pop, (AgeGroup)j, InfectionState::C) +
73  Po::get_from(pop, (AgeGroup)j, InfectionState::I) +
74  Po::get_from(pop, (AgeGroup)j, InfectionState::H) +
75  Po::get_from(pop, (AgeGroup)j, InfectionState::U) +
76  Po::get_from(pop, (AgeGroup)j, InfectionState::R); // without died people
77  ScalarType divNj = 1.0 / Nj; // precompute 1.0/Nj
78  ScalarType Si = Po::get_from(y, (AgeGroup)i, InfectionState::S);
79  ScalarType Cj = Po::get_from(pop, (AgeGroup)j, InfectionState::C);
80  ScalarType Ij = Po::get_from(pop, (AgeGroup)j, InfectionState::I);
81  return Si * cont_freq_eff * divNj * p.probabilities[i].get_infection_from_contact() *
82  (p.probabilities[j].get_carrier_infectability() * Cj +
83  risk_from_symptomatic * Ij);
84  });
85  }
86 
87  // Ei to Ci
88  this->add_flow(std::make_tuple((AgeGroup)i, InfectionState::E),
89  std::make_tuple((AgeGroup)i, InfectionState::C),
90  [i](Pa const& p, Eigen::Ref<const Eigen::VectorXd> /*pop*/, Eigen::Ref<const Eigen::VectorXd> y, double /*t*/) {
91  return Po::get_from(y, (AgeGroup)i, InfectionState::E) /
92  (2 * p.times[i].get_serialinterval() - p.times[i].get_incubation());
93  });
94 
95  // Ci to Ii
96  this->add_flow(std::make_tuple((AgeGroup)i, InfectionState::C),
97  std::make_tuple((AgeGroup)i, InfectionState::I),
98  [i](Pa const& p, Eigen::Ref<const Eigen::VectorXd> /*pop*/, Eigen::Ref<const Eigen::VectorXd> y, double /*t*/) {
99  double dummy_R3 = 0.5 / (p.times[i].get_incubation() - p.times[i].get_serialinterval());
100  double alpha = p.probabilities[i].get_asymp_per_infectious();
101  return ((1 - alpha) * dummy_R3) * Po::get_from(y, (AgeGroup)i, InfectionState::C);
102  });
103 
104  // Ci to Ri
105  this->add_flow(
106  std::make_tuple((AgeGroup)i, InfectionState::C), std::make_tuple((AgeGroup)i, InfectionState::R),
107  [i](Pa const& p, Eigen::Ref<const Eigen::VectorXd> /*pop*/, Eigen::Ref<const Eigen::VectorXd> y, double /*t*/) {
108  double alpha = p.probabilities[i].get_asymp_per_infectious();
109  return (alpha / p.times[i].get_infectious_asymp()) * Po::get_from(y, (AgeGroup)i, InfectionState::C);
110  });
111 
112  // Ii to Ri
113  this->add_flow(std::make_tuple((AgeGroup)i, InfectionState::I),
114  std::make_tuple((AgeGroup)i, InfectionState::R),
115  [i](Pa const& p, Eigen::Ref<const Eigen::VectorXd> /*pop*/, Eigen::Ref<const Eigen::VectorXd> y, double /*t*/) {
116  return Po::get_from(y, (AgeGroup)i, InfectionState::I) *
117  (1 - p.probabilities[i].get_hospitalized_per_infectious()) /
118  p.times[i].get_infectious_mild();
119  });
120 
121  // Ii to Hi
122  this->add_flow(
123  std::make_tuple((AgeGroup)i, InfectionState::I), std::make_tuple((AgeGroup)i, InfectionState::H),
124  [i](Pa const& p, Eigen::Ref<const Eigen::VectorXd> /*pop*/, Eigen::Ref<const Eigen::VectorXd> y, double /*t*/) {
125  return Po::get_from(y, (AgeGroup)i, InfectionState::I) *
126  p.probabilities[i].get_hospitalized_per_infectious() / p.times[i].get_home_to_hospitalized();
127  });
128 
129  // Hi to Ui
130  this->add_flow(std::make_tuple((AgeGroup)i, InfectionState::H),
131  std::make_tuple((AgeGroup)i, InfectionState::U),
132  [i](Pa const& p, Eigen::Ref<const Eigen::VectorXd> /*pop*/, Eigen::Ref<const Eigen::VectorXd> y, double /*t*/) {
133  ScalarType icu_occupancy = 0;
134  for (size_t j = 0; j < (size_t)AgeGroup::Count; ++j) {
135  icu_occupancy += Po::get_from(y, (AgeGroup)j, InfectionState::U);
136  }
137 
138  ScalarType prob_hosp2icu =
139  smoother_cosine(icu_occupancy, 0.90 * p.get_icu_capacity(), p.get_icu_capacity(),
140  p.probabilities[i].get_icu_per_hospitalized(), 0);
141 
142  return Po::get_from(y, (AgeGroup)i, InfectionState::H) * prob_hosp2icu /
143  p.times[i].get_hospitalized_to_icu();
144  });
145 
146  // Hi to Di
147  this->add_flow(
148  std::make_tuple((AgeGroup)i, InfectionState::H), std::make_tuple((AgeGroup)i, InfectionState::D),
149  [i](Pa const& p, Eigen::Ref<const Eigen::VectorXd> /*pop*/, Eigen::Ref<const Eigen::VectorXd> y, double /*t*/) {
150  ScalarType icu_occupancy = 0;
151  for (size_t j = 0; j < (size_t)AgeGroup::Count; ++j) {
152  icu_occupancy += Po::get_from(y, (AgeGroup)j, InfectionState::U);
153  }
154 
155  ScalarType prob_hosp2icu =
156  smoother_cosine(icu_occupancy, 0.90 * p.get_icu_capacity(), p.get_icu_capacity(),
157  p.probabilities[i].get_icu_per_hospitalized(), 0);
158  ScalarType prob_hosp2dead = p.probabilities[i].get_icu_per_hospitalized() - prob_hosp2icu;
159 
160  return Po::get_from(y, (AgeGroup)i, InfectionState::H) * prob_hosp2dead /
161  p.times[i].get_hospitalized_to_icu();
162  });
163 
164  // Hi to Ri
165  this->add_flow(
166  std::make_tuple((AgeGroup)i, InfectionState::H), std::make_tuple((AgeGroup)i, InfectionState::R),
167  [i](Pa const& p, Eigen::Ref<const Eigen::VectorXd> /*pop*/, Eigen::Ref<const Eigen::VectorXd> y, double /*t*/) {
168  return Po::get_from(y, (AgeGroup)i, InfectionState::H) *
169  (1 - p.probabilities[i].get_icu_per_hospitalized()) / p.times[i].get_hospitalized_to_home();
170  });
171 
172  // Ui to Ri
173  this->add_flow(std::make_tuple((AgeGroup)i, InfectionState::U),
174  std::make_tuple((AgeGroup)i, InfectionState::R),
175  [i](Pa const& p, Eigen::Ref<const Eigen::VectorXd> /*pop*/, Eigen::Ref<const Eigen::VectorXd> y, double /*t*/) {
176  return Po::get_from(y, (AgeGroup)i, InfectionState::U) *
177  (1 - p.probabilities[i].get_dead_per_icu()) / p.times[i].get_icu_to_home();
178  });
179 
180  // Ui to Di
181  this->add_flow(std::make_tuple((AgeGroup)i, InfectionState::U),
182  std::make_tuple((AgeGroup)i, InfectionState::D),
183  [i](Pa const& p, Eigen::Ref<const Eigen::VectorXd> /*pop*/, Eigen::Ref<const Eigen::VectorXd> y, double /*t*/) {
184  return Po::get_from(y, (AgeGroup)i, InfectionState::U) *
185  p.probabilities[i].get_dead_per_icu() / p.times[i].get_icu_to_dead();
186  });
187  }

198 void get_derivatives(Eigen::Ref<const Eigen::VectorXd> pop,
199  Eigen::Ref<const Eigen::VectorXd> y, double t,
200  Eigen::Ref<Eigen::VectorXd> dydt) const override
201 {
202  // alpha // percentage of asymptomatic cases
203  // beta // risk of infection from the infected symptomatic patients
204  // rho // hospitalized per infectious
205  // theta // icu per hospitalized
206  // delta // deaths per ICUs
207  // 0: S, 1: E, 2: C, 3: I, 4: H, 5: U, 6: R, 7: D
208  auto const& params = this->parameters;
209  AgeGroup n_agegroups = params.get_num_groups();
210 
211  ContactMatrixGroup const& contact_matrix = params.get<epi::ContactPatterns>();
212 
213  auto icu_occupancy = 0.0;
214  auto test_and_trace_required = 0.0;
215  for (auto i = AgeGroup(0); i < n_agegroups; ++i) {
216  auto dummy_R3 = 0.5 / (params.get<IncubationTime>()[i] - params.get<SerialInterval>()[i]);
217  test_and_trace_required += (1 - params.get<AsymptoticCasesPerInfectious>()[i]) * dummy_R3 *
218  this->populations.get_from(pop, {i, InfectionState::Carrier});
219  icu_occupancy += this->populations.get_from(pop, {i, InfectionState::ICU});
220  }
221 
222  for (auto i = AgeGroup(0); i < n_agegroups; i++) {
223 
224  size_t Si = this->populations.get_flat_index({i, InfectionState::Susceptible});
225  size_t Ei = this->populations.get_flat_index({i, InfectionState::Exposed});
226  size_t Ci = this->populations.get_flat_index({i, InfectionState::Carrier});
227  size_t Ii = this->populations.get_flat_index({i, InfectionState::Infected});
228  size_t Hi = this->populations.get_flat_index({i, InfectionState::Hospitalized});
229  size_t Ui = this->populations.get_flat_index({i, InfectionState::ICU});
230  size_t Ri = this->populations.get_flat_index({i, InfectionState::Recovered});
231  size_t Di = this->populations.get_flat_index({i, InfectionState::Dead});
232 
233  dydt[Si] = 0;
234  dydt[Ei] = 0;
235 
236  double dummy_R2 =
237  1.0 / (2 * params.get<SerialInterval>()[i] - params.get<IncubationTime>()[i]); // R2 = 1/(2SI-TINC)
238  double dummy_R3 =
239  0.5 / (params.get<IncubationTime>()[i] - params.get<SerialInterval>()[i]); // R3 = 1/(2(TINC-SI))
240 
241  for (auto j = AgeGroup(0); j < n_agegroups; j++) {
242  size_t Sj = this->populations.get_flat_index({j, InfectionState::Susceptible});
243  size_t Ej = this->populations.get_flat_index({j, InfectionState::Exposed});
244  size_t Cj = this->populations.get_flat_index({j, InfectionState::Carrier});
245  size_t Ij = this->populations.get_flat_index({j, InfectionState::Infected});
246  size_t Hj = this->populations.get_flat_index({j, InfectionState::Hospitalized});
247  size_t Uj = this->populations.get_flat_index({j, InfectionState::ICU});
248  size_t Rj = this->populations.get_flat_index({j, InfectionState::Recovered});
249 
250  //symptomatic are less well quarantined when testing and tracing is overwhelmed so they infect more people
251  auto risk_from_symptomatic = smoother_cosine(
252  test_and_trace_required, params.get<TestAndTraceCapacity>(), params.get<TestAndTraceCapacity>() * 5,
253  params.get<RiskOfInfectionFromSympomatic>()[j], params.get<MaxRiskOfInfectionFromSympomatic>()[j]);
254 
255  // effective contact rate by contact rate between groups i and j and damping j
256  double season_val =
257  (1 + params.get<epi::Seasonality>() *
258  sin(3.141592653589793 * (std::fmod((params.get<epi::StartDay>() + t), 365.0) / 182.5 + 0.5)));
259  double cont_freq_eff = season_val * contact_matrix.get_matrix_at(t)(static_cast<Eigen::Index>((size_t)i),
260  static_cast<Eigen::Index>((size_t)j));
261  double Nj = pop[Sj] + pop[Ej] + pop[Cj] + pop[Ij] + pop[Hj] + pop[Uj] + pop[Rj]; // without died people
262  double divNj = 1.0 / Nj; // precompute 1.0/Nj
263  double dummy_S = y[Si] * cont_freq_eff * divNj * params.get<InfectionProbabilityFromContact>()[i] *
264  (params.get<RelativeCarrierInfectability>()[j] * pop[Cj] +
265  risk_from_symptomatic * pop[Ij]);
266 
267  dydt[Si] -= dummy_S; // -R1*(C+beta*I)*S/N0
268  dydt[Ei] += dummy_S; // R1*(C+beta*I)*S/N0-R2*E
269  }
270 
271  // ICU capacity shortage is close
272  double prob_hosp2icu =
273  smoother_cosine(icu_occupancy, 0.90 * params.get<epi::ICUCapacity>(), params.get<epi::ICUCapacity>(),
274  params.get<ICUCasesPerHospitalized>()[i], 0);
275 
276  double prob_hosp2dead = params.get<ICUCasesPerHospitalized>()[i] - prob_hosp2icu;
277 
278 
279  dydt[Ei] -= dummy_R2 * y[Ei]; // only exchange of E and C done here
280  dydt[Ci] = dummy_R2 * y[Ei] -
281  ((1 - params.get<AsymptoticCasesPerInfectious>()[i]) * dummy_R3 +
282  params.get<AsymptoticCasesPerInfectious>()[i] / params.get<InfectiousTimeAsymptomatic>()[i]) *
283  y[Ci];
284  dydt[Ii] =
285  (1 - params.get<AsymptoticCasesPerInfectious>()[i]) * dummy_R3 * y[Ci] -
286  ((1 - params.get<HospitalizedCasesPerInfectious>()[i]) / params.get<InfectiousTimeMild>()[i] +
287  params.get<HospitalizedCasesPerInfectious>()[i] / params.get<HomeToHospitalizedTime>()[i]) *
288  y[Ii];
289  dydt[Hi] =
290  params.get<HospitalizedCasesPerInfectious>()[i] / params.get<HomeToHospitalizedTime>()[i] *
291  y[Ii] -
292  ((1 - params.get<ICUCasesPerHospitalized>()[i]) / params.get<HospitalizedToHomeTime>()[i] +
293  params.get<ICUCasesPerHospitalized>()[i] / params.get<HospitalizedToICUTime>()[i]) *
294  y[Hi];
295  dydt[Ui] = -((1 - params.get<DeathsPerHospitalized>()[i]) / params.get<ICUToHomeTime>()[i] +
296  params.get<DeathsPerHospitalized>()[i] / params.get<ICUToDeathTime>()[i]) *
297  y[Ui];
298  // add flow from hosp to icu according to potentially adjusted probability due to ICU limits
299  dydt[Ui] += prob_hosp2icu / params.get<HospitalizedToICUTime>()[i] * y[Hi];
300 
301  dydt[Ri] = params.get<AsymptoticCasesPerInfectious>()[i] / params.get<InfectiousTimeAsymptomatic>()[i] * y[Ci] +
302  (1 - params.get<HospitalizedCasesPerInfectious>()[i]) /
303  params.get<InfectiousTimeMild>()[i] * y[Ii] +
304  (1 - params.get<ICUCasesPerHospitalized>()[i]) /
305  params.get<HospitalizedToHomeTime>()[i] * y[Hi] +
306  (1 - params.get<DeathsPerHospitalized>()[i]) / params.get<ICUToHomeTime>()[i] * y[Ui];
307 
308  dydt[Di] = params.get<DeathsPerHospitalized>()[i] / params.get<ICUToDeathTime>()[i] * y[Ui];
309  // add potential, additional deaths due to ICU overflow
310  dydt[Di] += prob_hosp2dead / params.get<HospitalizedToICUTime>()[i] * y[Hi];
311  }
312 }

372  SecirSimulation(SecirModel const& model, double t0 = 0., double dt = 0.1)
373  : Base(model, t0, dt)
374  , m_t_last_npi_check(t0)

385  Eigen::Ref<Eigen::VectorXd> advance(double tmax)
386  {
387  auto& dyn_npis = this->get_model().parameters.template get<DynamicNPIsInfected>();
388  auto& contact_patterns = this->get_model().parameters.template get<ContactPatterns>();
389  if (dyn_npis.get_thresholds().size() > 0) {
390  auto t = Base::get_result().get_last_time();
391  const auto dt = dyn_npis.get_interval().get();
392 
393  while (t < tmax) {
394  auto dt_eff = std::min({dt, tmax - t, m_t_last_npi_check + dt - t});
395 
396  Base::advance(t + dt_eff);
397  t = t + dt_eff;
398 
399  if (floating_point_greater_equal(t, m_t_last_npi_check + dt)) {
400  auto inf_rel =
401  get_infections_relative(*this, t, this->get_result().get_last_value()) * dyn_npis.get_base_value();
402  auto exceeded_threshold = dyn_npis.get_max_exceeded_threshold(inf_rel);
403  if (exceeded_threshold != dyn_npis.get_thresholds().end() &&
404  (exceeded_threshold->first > m_dynamic_npi.first ||
405  t > double(m_dynamic_npi.second))) { //old npi was weaker or is expired
406  auto t_end = epi::SimulationTime(t + double(dyn_npis.get_duration()));
407  m_dynamic_npi = std::make_pair(exceeded_threshold->first, t_end);
408  epi::implement_dynamic_npis(contact_patterns.get_cont_freq_mat(), exceeded_threshold->second,
409  SimulationTime(t), t_end, [this](auto& g) {
410  return epi::make_contact_damping_matrix(g);
411  });
412  }
413 
414  m_t_last_npi_check = t;
415  }
416  }
417 
418  return this->get_result().get_last_value();
419  }
420  else {
421  return Base::advance(tmax);
422  }
423  }
424 
425 private:
426  double m_t_last_npi_check;
427  std::pair<double, SimulationTime> m_dynamic_npi = {-std::numeric_limits<double>::max(), epi::SimulationTime(0)};
428 };

467 template <class Base = Simulation<SecirModel>>
468 auto get_migration_factors(const SecirSimulation<Base>& sim, double /*t*/, const Eigen::Ref<const Eigen::VectorXd>& y)
469 {
470  auto& params = sim.get_model().parameters;
471  //parameters as arrays
472  auto& t_inc = params.template get<IncubationTime>().array().template cast<double>();
473  auto& t_ser = params.template get<SerialInterval>().array().template cast<double>();
474  auto& p_asymp = params.template get<AsymptoticCasesPerInfectious>().array().template cast<double>();
475  auto& p_inf = params.template get<RiskOfInfectionFromSympomatic>().array().template cast<double>();
476  auto& p_inf_max = params.template get<MaxRiskOfInfectionFromSympomatic>().array().template cast<double>();
477  //slice of carriers
478  auto y_car = slice(y, {Eigen::Index(InfectionState::Carrier), Eigen::Index(size_t(params.get_num_groups())),
479  Eigen::Index(InfectionState::Count)});
480 
481  //compute isolation, same as infection risk from main model
482  auto R3 = 0.5 / (t_inc - t_ser);
483  auto test_and_trace_required = ((1 - p_asymp) * R3 * y_car.array()).sum();
484  auto risk_from_symptomatic =
485  smoother_cosine(test_and_trace_required, double(params.template get<TestAndTraceCapacity>()),
486  params.template get<TestAndTraceCapacity>() * 5, p_inf.matrix(), p_inf_max.matrix());
