120 
121  auto num_runs = ensemble_params.size();
122  auto num_nodes = ensemble_params[0].size();

124 
125  std::vector<double> single_element_ensemble(num_runs);

127  // lamda function that calculates the percentile of a single paramter
128  std::vector<Model> percentile(num_nodes, Model((int)num_groups));
129  auto param_percentil = [&ensemble_params, p, num_runs, &percentile](auto n, auto get_param) mutable {
130  std::vector<double> single_element(num_runs);
131  for (size_t run = 0; run < num_runs; run++) {
132  auto const& params = ensemble_params[run][n];
133  single_element[run] = get_param(params);
134  }
135  std::sort(single_element.begin(), single_element.end());
136  auto& new_params = get_param(percentile[n]);
137  new_params = single_element[static_cast<size_t>(num_runs * p)];
138  };
139 
140  for (size_t node = 0; node < num_nodes; node++) {
141  for (auto i = AgeGroup(0); i < AgeGroup(num_groups); i++) {
142  //Population

200  for (size_t run = 0; run < num_runs; run++) {
201

203  single_element_ensemble[run] = params.parameters.template get<epi::ICUCapacity>() * params.populations.get_total();
204  }
205  std::sort(single_element_ensemble.begin(), single_element_ensemble.end());
206  percentile[node].parameters.template set<epi::ICUCapacity>(single_element_ensemble[static_cast<size_t>(num_runs * p)]);
207  }
