31 """! Gets the matrix of commuter migration patterns and all additional helper variables.

71def get_key_and_population_lists(setup_dict): 

72 """! Get list of regional key identifiers and population sizes for counties. 

73 @param setup_dict Dictionary with necessary values for 'counties', 'num_counties', 'num_govregions','path', 

74 'abs_tol', 'rel_tol' 

75 

76 @return countykey_list List of county regional keys 

77 @return countypop_list List of populations per counties. 

78 @return govkey_list List of governing regions regional keys. 

79 """ 

80 # get and store all regional (county) identifiers in a list; store county populations accordingly 

81 # get a list of governing regions 

82 countykey_list = [] 

83 countypop_list = [] 

84 govkey_list = [] 

85 counties = setup_dict['counties'] 

86 for i in range(0, counties.shape[0]): 

87 # regional county identifieres (5 numbers) 

88 if len(str(counties.iloc[i][0])) == 5 and counties.iloc[i][0].isdigit(): 

89 countykey_list.append(counties.iloc[i][0]) 

90 countypop_list.append(counties.iloc[i][5]) 

91 # government region keys (2 or 3 numbers) 

92 elif i < counties.shape[0] - 1 and len(str(counties.iloc[i][0])) < len(str(counties.iloc[i + 1][0])): 

93 # workaround for old gov. regions and Saxony 

94 if (not str(counties.iloc[i][1]).startswith('früher') and not str(counties.iloc[i][1]).startswith( 

95 'Direktion')): 

96 # only take those keys which have less numbers than the key in the next row 

97 if len(str(counties.iloc[i][0])) != 4 and len(str(counties.iloc[i + 1][0])) == 5: 

98 # where string length is not 4 and next key has length four 

99 # these rows correspond to 'local government' regions (except for BW, RP and Saxony) 

100 govkey_list.append(counties.iloc[i][0]) 

101 

102 elif i < counties.shape[0] - 2: 

103 if len(str(counties.iloc[i][0])) == 3 and len(str(counties.iloc[i + 2][0])) == 5: 

104 # workaround for BW; 'government regions' are again divided but do not appear as such in 

105 # documents of the Arbeitsagentur 

106 govkey_list.append(counties.iloc[i][0]) 

107 

108 if len(str(counties.iloc[i][0])) == 2 and len(str(counties.iloc[i + 2][0])) == 5: 

109 # workaround for RP and Saxony; 

110 if str(counties.iloc[i + 1][1]).startswith('früher'): 

111 # workaround for RP; 'government regions' were dissolved 

112 govkey_list.append(counties.iloc[i][0]) 

113 elif str(counties.iloc[i + 1][1]).startswith('Direktion'): 

114 # workaround for Saxony; 'Direktionsbezirke' not referred in commuter migration 

115 govkey_list.append(counties.iloc[i][0]) 

116 

117 if len(govkey_list) != setup_dict['num_govregions']: 

118 print('Error. Number of government regions wrong. Having', len(govkey_list), 'instead of', 

119 setup_dict['num_govregions']) 

120 

121 return countykey_list, countypop_list, govkey_list

136def map_keys_to_numlists(setup_dict, countykey_list=None, govkey_list=None): 

137 """! Creates hash maps from from county regional keys and keys of its governing regions to numbered lists. 

138 

139 Keyword arguments: 

140 @param setup_dict Dictionary with necessary values for 'counties', 'num_counties', 'num_govregions','path', 

141 'abs_tol', 'rel_tol' 

142 @param countykey_list List of county regional keys. 

143 @param govkey_list List of governing regions regional keys. 

144 

145 @return countykey2numlist Hash map from county regional keys to numbered list. 

146 @return govkey2numlist Hash map from governing region regional keys to numbered list. 

147 """ 

148 # create a hashmap from sorted regional identifiers (01001 - ...) to 0 - num_counties 

149 if countykey_list is None or govkey_list is None: 

150 countykey_list, _, govkey_list = get_key_and_population_lists(setup_dict) 

151 verify_sorted(countykey_list) 

152 

153 countykey2numlist = collections.OrderedDict() 

154 i = 0 

155 for index in countykey_list: 

156 countykey2numlist[index] = i 

157 i += 1 

158 

159 if i != setup_dict['num_counties']: 

160 print("Error. Number of counties wrong.") 

161 

162 # create a hash map from sorted gov keys to local list 

163 govkey2numlist = collections.OrderedDict() 

164 i = 0 

165 for index in govkey_list: 

166 govkey2numlist[index] = i 

167 i += 1 

168 

169 if i != setup_dict['num_govregions']: 

170 print("Error. Number of governing regions wrong.") 

171 

172 return countykey2numlist, govkey2numlist

175def assign_geographical_entities(setup_dict, countykey_list=None, govkey_list=None): 

176 """! Assigns counties to governing regions based on key comparison and creates list of governing regions per state. 

177 

178 Only works with sorted key lists. 

179 

180 Keyword arguments: 

181 @param setup_dict dictionary with necessary values 

182 @param countykey_list List of county regional keys. 

183 @param govkey_list List of governing regions regional keys. 

184 

185 @return countykey2govkey Hash map from county regional keys to governing region regional keys. 

186 @return countykey2localnumlist Hash map from county regional keys to local numbered list (per governing region). 

187 @return gov_county_table Table of county regional keys per governing region. 

188 @return state_gov_table Table of governing region regional keys per federal state. 

189 """ 

190 

191 if govkey_list is None or countykey_list is None: 

192 countykey_list, _, govkey_list = get_key_and_population_lists(setup_dict) 

193 

194 verify_sorted(countykey_list) 

195 

196 # Create list of government regions with lists of counties that belong to them and list of states with government 

197 # regions that belong to them; only works with sorted lists of keys. 

198 gov_county_table = [] 

199 

200 gov_index = 0 

201 col_index = 0 

202 col_list = [] 

203 

204 for i in range(0, len(countykey_list)): 

205 

206 # check for belonging to currently considered government region 

207 if str(countykey_list[i]).startswith(str(govkey_list[gov_index])): 

208 col_list.append(countykey_list[i]) # add county to current government region 

209 col_index += 1 

210 # go to next government region 

211 if i < len(countykey_list) - 1 and (not str(countykey_list[i + 1]).startswith(str(govkey_list[gov_index]))): 

212 # add government region to full table 

213 gov_county_table.append(col_list) 

214 col_list = [] 

215 gov_index += 1 

216 col_index = 0 

217 # add last government region 

218 gov_county_table.append(col_list) 

219 

220 if len(gov_county_table) != setup_dict['num_govregions']: 

221 print('Error. Number of government regions wrong.') 

222 

223 # create a unique hash map from county key to its government region and 

224 # a global key to local (in gov region) key ordering 

225 countykey2govkey = collections.OrderedDict() 

226 countykey2localnumlist = collections.OrderedDict() 

227 for i in range(0, len(gov_county_table)): 

228 for j in range(0, len(gov_county_table[i])): 

229 countykey2govkey[gov_county_table[i][j]] = i 

230 countykey2localnumlist[gov_county_table[i][j]] = j 

231 

232 # create government regions list per state 

233 state_gov_table = [] 

234 

235 state_id = 1 

236 state_govlist_loc = [] 

237 for i in range(0, len(govkey_list)): 

238 

239 if str(int(govkey_list[i])).startswith(str(state_id)): 

240 state_govlist_loc.append(govkey_list[i]) 

241 

242 if i + 1 < len(govkey_list) and not str(int(govkey_list[i + 1])).startswith(str(state_id)): 

243 state_id += 1 

244 state_gov_table.append(state_govlist_loc) 

245 state_govlist_loc = [] 

246 # add last state's list 

247 state_gov_table.append(state_govlist_loc) 

248 

249 return countykey2govkey, countykey2localnumlist, gov_county_table, state_gov_table
