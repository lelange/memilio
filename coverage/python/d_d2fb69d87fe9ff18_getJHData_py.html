34def get_jh_data(read_data=dd.defaultDict['read_data'], 

35 file_format=dd.defaultDict['file_format'], 

36 out_folder=dd.defaultDict['out_folder'], 

37 no_raw=dd.defaultDict['no_raw']): 

38 """! Download data from John Hopkins University 

39 

40 Data is either downloaded and afterwards stored or loaded from a stored filed. 

41 The file is "FullData_JohnHopkins.json" 

42 

43 Working with the data includes 

44 - rename columns such that "/" is deleted, e.g Country/Region becomes CountryRegion 

45 - data of all countries together are written to a file 

46 - download the data from following countries in a separate file 

47 and are stored in the according folders with the country name 

48 - Germany, SouthKorea, Spain, France, Italy, US, China 

49 - furthermore, all countries, for which provinces are added, are written to a file 

50 

51 @param read_data False [Default] or True. Defines if data is read from file or downloaded. 

52 @param file_format File format which is used for writing the data. Default defined in defaultDict. 

53 @param out_folder Path to folder where data is written in folder out_folder/Germany. 

54 @param no_raw True or False [Default]. Defines if unchanged raw data is saved or not. 

55 """ 

56 

57 filename = "FullData_JohnHopkins" 

58 

59 if read_data: 

60 file_in = os.path.join(out_folder, filename + ".json") 

61 # if once dowloaded just read json file 

62 try: 

63 df = pandas.read_json(file_in) 

64 except ValueError: 

65 exit_string = "Error: The file: " + file_in + " does not exist. Call program without -r flag to get it." 

66 sys.exit(exit_string) 

67 else: 

68 # Get data: 

69 # https://raw.githubusercontent.com/datasets/covid-19/master/data/time-series-19-covid-combined.csv 

70 df = gd.loadCsv('time-series-19-covid-combined', 

71 apiUrl='https://raw.githubusercontent.com/datasets/covid-19/master/data/') 

72 

73 # convert "Datenstand" to real date: 

74 # df.Datenstand = pandas.to_datetime( df.Datenstand, format='%d.%m.%Y, %H:%M Uhr').dt.tz_localize 

75 # ('Europe/Berlin') 

76 

77 # output data to not always download it 

78 if not no_raw: 

79 gd.write_dataframe(df, out_folder, filename, "json") 

80 

81 df.rename({'Country/Region': 'CountryRegion', 'Province/State': 'ProvinceState'}, axis=1, inplace=True) 

82 print("Available columns:", df.columns) 

83 

84 # Change "Korea, South" to SouthKorea 

85 df.loc[df['CountryRegion'] == "Korea, South", ['CountryRegion']] = 'SouthKorea' 

86 

87 # Generate folders if needed 

88 directory_ger = os.path.join(out_folder, 'Germany/') 

89 directory_es = os.path.join(out_folder, 'Spain/') 

90 directory_fr = os.path.join(out_folder, 'France/') 

91 directory_it = os.path.join(out_folder, 'Italy/') 

92 directory_us = os.path.join(out_folder, 'US/') 

93 directory_rok = os.path.join(out_folder, 'SouthKorea/') 

94 directory_prc = os.path.join(out_folder, 'China/') 

95 

96 # dictionary of countries 

97 countries = { 

98 "Germany": directory_ger, 

99 "Spain": directory_es, 

100 "France": directory_fr, 

101 "Italy": directory_it, 

102 "US": directory_us, 

103 "SouthKorea": directory_rok, 

104 "China": directory_prc, 

105 } 

106 

107 ########### Countries ########################## 

108 

109 gb = df.groupby(['CountryRegion', 'Date']).agg({"Confirmed": sum, "Recovered": sum, "Deaths": sum}) 

110 

111 gd.write_dataframe(gb.reset_index(), out_folder, "all_countries_jh", file_format) 

112 

113 for key in countries: 

114 # get data for specific countries 

115 gb_country = gb.reset_index()[gb.reset_index()["CountryRegion"] == key] 

116 dir_country = countries[key] 

117 gd.check_dir(dir_country) 

118 gd.write_dataframe(gb_country, dir_country, "whole_country_" + key + "_jh", file_format) 

119 

120 # Check what about external provinces. Should they be added? 

121 

122 ################ Countries with given States ################################ 

123 

124 # Get all countries where States are given 

125 dfD = df[~df["ProvinceState"].isnull()] 

126 

127 gb = dfD.groupby(['CountryRegion', 'ProvinceState', 'Date']).agg( 

128 {"Confirmed": sum, "Recovered": sum, "Deaths": sum}) 

129 

130 gd.write_dataframe(gb.reset_index(), out_folder, "all_provincestate_jh", file_format) 

131
