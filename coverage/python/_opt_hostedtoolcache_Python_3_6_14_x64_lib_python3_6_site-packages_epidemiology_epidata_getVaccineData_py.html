30# Downloads vaccine data from RKI 

31def download_vaccine_data(): 

32 # get rki vaccine data 

33 url = 'https://www.rki.de/DE/Content/InfAZ/N/Neuartiges_Coronavirus/Daten/Impfquotenmonitoring.xlsx?__blob=publicationFile' 

34 header = {'User-Agent': 'Mozilla/5.0'} 

35 r = requests.get(url, headers=header) 

36 with io.BytesIO(r.content) as fh: 

37 df = pandas.io.excel.ExcelFile(fh, engine='openpyxl') 

38 sheet_names = df.sheet_names 

39 vaccine = pandas.read_excel(df, sheet_name=sheet_names[1], header=[1])[2:18].drop(['Unnamed: 1', 'Unnamed: 10'], axis=1) 

40 name = sheet_names[2].replace('.', '_', 2).replace('.', '', 1)[-8:] 

41 name = 'vaccine_data_' + name 

42 

43 return vaccine, name

45# gets rki vaccine monitoring data for all states and extrapolates the values for counties according to their population 

46# Missing ratio values for the two different age groups are also estimated 

47def get_vaccine_data(read_data=dd.defaultDict['read_data'], 

48 file_format=dd.defaultDict['file_format'], 

49 out_folder=dd.defaultDict['out_folder'], 

50 no_raw=dd.defaultDict['no_raw']): 

51 

52 vaccine, name = download_vaccine_data() 

53 

54 col_names = vaccine.columns 

55 vaccine = vaccine.replace('-', np.nan) 

56 vaccine[col_names[5]].fillna(vaccine[col_names[5]].mean(), inplace=True) 

57 

58 # get population data for all countys 

59 population = getPopulationData.get_age_population_data(write_df=False) 

60 

61 # create empty dataframe 

62 new_df = pandas.DataFrame([], columns=['ID_County', 'Administrated_Vaccines', 'First_Shot', 'Full_Vaccination', 

63 'Ratio_All', 'Ratio_Young', 'Ratio_Old']) 

64 

65 # loop over all states 

66 for state, i in zip(vaccine['Unnamed: 0'], range(len(vaccine))): 

67 # get population data of current state 

68 pop_state = population[(population['ID_County'].values/1000).astype(int) == int(state)] 

69 

70 # get county ids 

71 id_state = pop_state.ID_County 

72 

73 # compute population ratio between countys and their state 

74 ratio = pop_state['Total'].values/sum(pop_state['Total']) 

75 

76 # scale vaccine values to countys 

77 state_values = np.outer(ratio,vaccine.values[i, 1:4]) 

78 

79 # store scaled values in dataframe 

80 data = np.zeros((len(ratio), len(vaccine.values[i,:7]))) 

81 data[:,0] = id_state 

82 data[:,1:4] = state_values 

83 data[:,4] =vaccine.values[i, 4] 

84 data[:,5] =vaccine.values[i, 5] 

85 if str(vaccine.values[i, 6]) == 'nan': 

86 pop_columns = ['<3 years', '3-5 years', '6-14 years', '15-17 years', '18-24 years', 

87 '25-29 years', '30-39 years', '40-49 years', '50-64 years', 

88 '65-74 years', '>74 years'] 

89 data[:, 6] = np.divide(pop_state['Total'].values*vaccine.values[i, 4] \ 

90 - (np.sum([pop_state[x].values for x in pop_columns[:8]]) + pop_state[pop_columns[8]].values*2/3) * vaccine.values[i, 5],\ 

91 pop_state[pop_columns[8]].values*1/3 + np.sum(pop_state[x].values for x in pop_columns[9:])) 

92 else: 

93 data[:, 6] = vaccine.values[i,6] 

94 state_df = pandas.DataFrame(data, columns=new_df.columns) 

95 new_df = new_df.append(state_df) 

96 

97 directory = out_folder 

98 directory = os.path.join(directory, 'Germany/') 

99 gd.check_dir(directory) 

100 

101 gd.write_dataframe(new_df, directory, name, file_format) 

102 if file_format=='json_timeasstring': 

103 file_format = 'json' 

104 print('file written to:', directory + name + '.' + file_format)
